4. Implémenter handler onCardMove :
```typescript
const handleCardMove = async (cardId: string, fromCol: string, toCol: string) => {
  try {
    // Optimistic update
    setOpportunites(prev =>
      prev.map(o => o.id === cardId ? { ...o, status: toCol as any } : o)
    )

    // API call
    await fetch(`/api/advisor/opportunites/${cardId}`, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ status: toCol }),
    })

    toast({
      title: 'Opportunité mise à jour',
      description: `Statut changé vers "${getStageLabel(toCol)}"`,
      variant: 'success',
    })
  } catch (error) {
    // Rollback
    await loadOpportunites()
    toast({
      title: 'Erreur',
      description: 'Impossible de modifier le statut',
      variant: 'destructive',
    })
  }
}
```

const renderOpportuniteCard = (opp: OpportuniteKanbanData, isDragging: boolean) => (
