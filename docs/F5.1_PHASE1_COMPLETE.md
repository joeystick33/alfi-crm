```prisma
model Invoice {
  id              String        @id @default(cuid())
  cabinetId       String
  invoiceNumber   String        // INV-2024-001
  clientId        String
  conseillerId    String
  amountHT        Decimal       @db.Decimal(12, 2)
  tva             Decimal       @db.Decimal(5, 2)
  amountTTC       Decimal       @db.Decimal(12, 2)
  status          InvoiceStatus @default(DRAFT)
  issueDate       DateTime      @default(now())
  dueDate         DateTime
  paidDate        DateTime?
  description     String?       @db.Text
  notes           String?       @db.Text
  termsAndConditions String?    @db.Text
  projectId       String?       // Lié à un projet
  opportunityId   String?       // Lié à une opportunité
  pdfUrl          String?
  pdfGeneratedAt  DateTime?
  createdAt       DateTime      @default(now())
  updatedAt       DateTime      @updatedAt
  
  // Relations
  cabinet         Cabinet
  client          Client
  conseiller      User          @relation("ConseillerInvoices")
  items           InvoiceItem[]
  payments        Payment[]
  project         Projet?
  opportunity     Opportunite?
  commissions     Commission[]
  
  @@unique([cabinetId, invoiceNumber])
  @@index([cabinetId, clientId])
  @@index([cabinetId, conseillerId])
  @@index([cabinetId, status])
  @@index([issueDate])
  @@index([dueDate])
}
```

#### 5. Commission (16 champs)
```prisma
model Commission {
  id          String           @id @default(cuid())
  cabinetId   String
  apporteurId String
  clientId    String
  invoiceId   String?
  amount      Decimal          @db.Decimal(12, 2)
  rate        Decimal          @db.Decimal(5, 2)
  type        CommissionType   @default(PERCENTAGE)
  status      CommissionStatus @default(PENDING)
  paidAt      DateTime?
  description String?          @db.Text
  notes       String?          @db.Text
  createdAt   DateTime         @default(now())
  updatedAt   DateTime         @updatedAt
  
  // Relations
  apporteur   Apporteur        @relation(fields: [apporteurId], onDelete: Cascade)
  client      Client           @relation(fields: [clientId])
  invoice     Invoice?         @relation(fields: [invoiceId], onDelete: SetNull)
  
  @@index([cabinetId, apporteurId])
  @@index([cabinetId, clientId])
  @@index([cabinetId, status])
  @@index([cabinetId, invoiceId])
}
```
