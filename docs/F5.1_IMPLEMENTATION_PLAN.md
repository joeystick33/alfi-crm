## 🏗️ ARCHITECTURE F5.1

### Stratégie Prospects

**Option A** : Utiliser Client.status = PROSPECT ✅ **RECOMMANDÉ**
- Avantages :
  - Cohérent avec schema existant
  - Pas de duplication données
  - Conversion prospect → client = update status
  - Relations déjà disponibles (conseiller, opportunités, tâches)
- Inconvénients :
  - Champs Client peut-être trop riches pour prospects

**Option B** : Créer table Prospect séparée
- Avantages :
  - Champs spécifiques prospects
  - Séparation logique
- Inconvénients :
  - Duplication avec Client
  - Migration prospect → client complexe
  - Relations à refaire

**Décision** : **Option A** - Utiliser Client.status = PROSPECT

### Architecture Facturation

**Modèle Invoice requis** :
```prisma
model Invoice {
  id                  String          @id @default(cuid())
  cabinetId           String
  invoiceNumber       String          // Numéro unique: INV-2024-001
  
  // Client facturé
  clientId            String
  client              Client          @relation(fields: [clientId], references: [id])
  
  // Conseiller
  conseillerId        String
  conseiller          User            @relation("ConseillerInvoices", fields: [conseillerId], references: [id])
  
  // Montants
  amountHT            Decimal         @db.Decimal(10, 2)
  tva                 Decimal         @db.Decimal(5, 2) // %
  amountTTC           Decimal         @db.Decimal(10, 2)
  
  // Statut
  status              InvoiceStatus   @default(DRAFT)
  
  // Dates
  issueDate           DateTime        @default(now())
  dueDate             DateTime
  paidDate            DateTime?
  
  // Description
  description         String?         @db.Text
  notes               String?         @db.Text
  
  // Lignes facture
  items               InvoiceItem[]
  
  // Relations optionnelles
  projectId           String?
  project             Projet?         @relation(fields: [projectId], references: [id])
  
  opportunityId       String?
  opportunity         Opportunite?    @relation(fields: [opportunityId], references: [id])
  
  // Paiements
  payments            Payment[]
  
  // PDF
  pdfUrl              String?
  
  // Metadata
  createdAt           DateTime        @default(now())
  updatedAt           DateTime        @updatedAt
  
  @@unique([cabinetId, invoiceNumber])
  @@index([cabinetId, clientId])
  @@index([cabinetId, conseillerId])
  @@index([cabinetId, status])
  @@index([issueDate])
  @@map("invoices")
}

model InvoiceItem {
  id          String    @id @default(cuid())
  invoiceId   String
  invoice     Invoice   @relation(fields: [invoiceId], references: [id], onDelete: Cascade)
  
  description String
  quantity    Decimal   @db.Decimal(10, 2)
  unitPrice   Decimal   @db.Decimal(10, 2)
  totalHT     Decimal   @db.Decimal(10, 2)
  tva         Decimal   @db.Decimal(5, 2) // %
  totalTTC    Decimal   @db.Decimal(10, 2)
  
  @@index([invoiceId])
  @@map("invoice_items")
}

model Payment {
  id          String        @id @default(cuid())
  cabinetId   String
  invoiceId   String
  invoice     Invoice       @relation(fields: [invoiceId], references: [id])
  
  amount      Decimal       @db.Decimal(10, 2)
  method      PaymentMethod @default(BANK_TRANSFER)
  reference   String?
  paidAt      DateTime      @default(now())
  notes       String?       @db.Text
  
  createdAt   DateTime      @default(now())
  
  @@index([cabinetId, invoiceId])
  @@map("payments")
}

enum InvoiceStatus {
  DRAFT
  SENT
  PAID
  OVERDUE
  CANCELLED
}

enum PaymentMethod {
  BANK_TRANSFER
  CHECK
  CASH
  CREDIT_CARD
  OTHER
}
```

### Architecture Apporteurs

**Modèle Apporteur requis** :
```prisma
model Apporteur {
  id            String   @id @default(cuid())
  cabinetId     String
  cabinet       Cabinet  @relation(fields: [cabinetId], references: [id], onDelete: Cascade)
  
  // Identité
  type          ApporteurType @default(INDIVIDUAL) // Personne physique ou morale
  firstName     String?
  lastName      String?
  companyName   String?
  
  // Contact
  email         String
  phone         String?
  address       String?
  city          String?
  postalCode    String?
  country       String   @default("France")
  
  // Professionnel
  siret         String?
  tvaNumber     String?
  
  // Commission
  commissionRate Decimal  @db.Decimal(5, 2) // %
  commissionType CommissionType @default(PERCENTAGE)
  
  // Relations
  clients       Client[] @relation("ApporteurClients")
  commissions   Commission[]
  
  // Statut
  isActive      Boolean  @default(true)
  
  // Notes
  notes         String?  @db.Text
  
  // Metadata
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
  
  @@unique([cabinetId, email])
  @@index([cabinetId, isActive])
  @@map("apporteurs")
}

model Commission {
  id            String     @id @default(cuid())
  cabinetId     String
  
  apporteurId   String
  apporteur     Apporteur  @relation(fields: [apporteurId], references: [id])
  
  clientId      String
  client        Client     @relation(fields: [clientId], references: [id])
  
  invoiceId     String?
  invoice       Invoice?   @relation(fields: [invoiceId], references: [id])
  
  amount        Decimal    @db.Decimal(10, 2)
  rate          Decimal    @db.Decimal(5, 2) // %
  
  status        CommissionStatus @default(PENDING)
  
  paidAt        DateTime?
  notes         String?    @db.Text
  
  createdAt     DateTime   @default(now())
  updatedAt     DateTime   @updatedAt
  
  @@index([cabinetId, apporteurId])
  @@index([cabinetId, clientId])
  @@index([cabinetId, status])
  @@map("commissions")
}

enum ApporteurType {
  INDIVIDUAL   // Personne physique
  COMPANY      // Société
}

enum CommissionType {
  PERCENTAGE   // % du CA
  FIXED        // Montant fixe
}

enum CommissionStatus {
  PENDING
  APPROVED
  PAID
  CANCELLED
}
```
